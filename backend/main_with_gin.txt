package main

import (
	"fmt"
	"log"
	"net/http"

	"github.com/gin-gonic/gin"
)

func checkError(err error) {
	if err != nil {
		log.Print("[[main]]]: ", err)

	}
}

func setupRouter() *gin.Engine {
	r := gin.Default()
	r.LoadHTMLGlob("templates/*")

	r.GET("/index", func(c *gin.Context) {
		c.HTML(http.StatusOK, "index.html", gin.H{
			"title": "Chat Widget",
		})
	})
	api := r.Group("/api")
	api.POST("/get_user_payload", getUserPayload)

	return r
}

func getUserPayload(c *gin.Context) {
	var payload struct {
		Message string `json:"message"`
	}
	if err := c.ShouldBindJSON(&payload); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": "invalid payload"})
		return
	}
	log.Print("[trying to get AI answer with payload] : ", payload.Message)

	request := fmt.Sprint("extract only company/curency name as on stock market, the answer will be only one word this name company/curency in English", payload.Message)
	//TODO сделать нормальный реквест к китайцу, а то он хренью страдает
	ai_answer, err := src.Ai_send_request("assistant", request)
	checkError(err)
	log.Print("[final answer]:", ai_answer)

	if err == nil {
		stocks_answer, err := src.Stocks_get_info(ai_answer)
		checkError(err)
		log.Print("[stocks_answer]: ", stocks_answer)
	}

	//сюда ниже итоговый реплай
	c.JSON(http.StatusOK, gin.H{
		"reply": "ok",
	})
}

func main() {
	r := setupRouter()
	r.Static("/static", "./static")
	r.Run(":8080")
}
